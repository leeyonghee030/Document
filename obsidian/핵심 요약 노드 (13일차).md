13일차: 배열 심화

키워드: 배열과 메서드 (심화: map, filter, find, sort, reduce)

실습: 객체들이 담긴 배열(예: 상품 목록)에서 가격이 5000원 이상인 것만 filter하고, 이름을 대문자로 바꾸는 map 연결해서 사용해보기.


// 1. 객체들이 담긴 배열 만들기
let products = [
  { name: 'banana', price: 5000 },
  { name: 'Apple', price: 3000 },
  { name: 'Kiwi', price: 2000 },
  { name: 'Grape', price: 6000 }
];

// 2. filter와 map 연결해서 사용하기
let result = products
  .filter(item => item.price >= 5000)      // 가격이 5000원 이상인 것만 골라내기
  .map(item => item.name.toUpperCase());   // 골라낸 것들의 이름을 대문자로 바꾸기

console.log(result); // 출력: ["BANANA", "GRAPE"]
